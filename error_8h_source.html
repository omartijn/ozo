410  std::string message(int value) const override {
411  switch (code(value)) {

413  return "no error";
414  case pq_connection_start_failed:
415  return "pq_connection_start_failed - PQConnectionStart function failed";
416  case pq_socket_failed:
417  return "pq_socket_failed - PQSocket returned -1 as fd - it seems like there is no connection";
418  case pq_connection_status_bad:
419  return "pq_connection_status_bad - PQstatus returned CONNECTION_BAD";
420  case pq_connect_poll_failed:
421  return "pq_connect_poll_failed - PQconnectPoll function failed";
422  case oid_type_mismatch:
423  return "no conversion possible from oid to user-supplied type";
424  case unexpected_eof:
425  return "unexpected EOF while data read";
426  case pg_send_query_params_failed:
427  return "pg_send_query_params_failed - PQsendQueryParams function failed";
428  case pg_consume_input_failed:
429  return "pg_consume_input_failed - PQconsumeInput function failed";
430  case pg_set_nonblocking_failed:
431  return "pg_set_nonblocking_failed - PQsetnonblocking function failed";
432  case pg_flush_failed:
433  return "pg_flush_failed - PQflush function failed";
434  case bad_result_process:
435  return "bad_result_process - error while processing or converting result from the database";
436  case no_sql_state_found:
437  return "no_sql_state_found - no sql state has been found in database query execution error reply";
438  case result_status_unexpected:
439  return "result_status_unexpected - got unexpected status from query result";
440  case result_status_empty_query:
441  return "result_status_empty_query - the string sent to the server was empty";
442  case result_status_bad_response:
